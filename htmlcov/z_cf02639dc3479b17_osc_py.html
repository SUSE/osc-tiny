232 def request(self, url: str, method: str = "GET", stream: bool = False, 

233 data: typing.Optional[ParamsType] = None, 

234 params: typing.Optional[ParamsType] = None, 

235 raise_for_status: bool = True, timeout: typing.Optional[int] = None) \ 

236 -> typing.Optional[Response]: 

237 """ 

238 Perform HTTP(S) request 

239 

240 ``data`` is URL-encoded and passed on as GET parameters. If ``data`` is 

241 a dictionary and contains a key ``comment``, this value is passed on as 

242 a POST parameter. 

243 

244 if ``raise_for_status`` is True, the used ``requests`` framework will 

245 raise an exception for occured errors. 

246 

247 .. versionchanged:: 0.1.2 

248 Added parameters `raise_for_status` 

249 

250 .. versionchanged:: 0.1.3 

251 

252 * Added parameter `timeout` 

253 * Transfer data as GET parameters (except for comments and texts) 

254 

255 .. versionchanged:: 0.1.5 

256 Retry sending the request, if the remote host disconnects 

257 

258 .. versionadded:: 0.1.7 

259 Added parameter `params` 

260 

261 .. versionchanged:: 0.5.0 

262 Added logging of request/response 

263 

264 :param url: Full URL 

265 :param method: HTTP method 

266 :param stream: Delayed access, see `Body Content Workflow`_ 

267 :param data: Data to be included as POST parameters in request 

268 :param params: Additional GET parameters to be included in request 

269 :param raise_for_status: See `requests.Response.raise_for_status`_ 

270 :param timeout: Request timeout. See `Timeouts`_ 

271 :return: :py:class:`requests.Response` 

272 

273 .. _Body Content Workflow: 

274 http://docs.python-requests.org/en/master/user/advanced/ 

275 #body-content-workflow 

276 

277 .. _requests.Response.raise_for_status: 

278 https://2.python-requests.org/en/master/api/ 

279 #requests.Response.raise_for_status 

280 

281 .. _Timeouts: 

282 https://2.python-requests.org/en/master/user/advanced/#timeouts 

283 """ 

284 timeout = timeout or self.default_timeout 

285 

286 req = Request( 

287 method, 

288 url.replace("#", quote("#")).replace("?", quote("?")), 

289 data=self.handle_params(url=url, method=method, params=data), 

290 params=self.handle_params(url=url, method=method, params=params) 

291 ) 

292 prepped_req = self.session.prepare_request(req) 

293 prepped_req.headers['Content-Type'] = "application/octet-stream" 

294 prepped_req.headers['Accept'] = "application/xml" 

295 settings = self.session.merge_environment_settings( 

296 prepped_req.url, {}, None, None, None 

297 ) 

298 settings["stream"] = stream 

299 if timeout: 

300 settings["timeout"] = timeout 

301 

302 try: 

303 response = self.session.send(prepped_req, **settings) 

304 except _ConnectionError as error: 

305 warnings.warn("Problem connecting to server: {}".format(error)) 

306 else: 

307 if raise_for_status: 

308 response.raise_for_status() 

309 return response 

310 

311 return None

335 def handle_params(self, url: str, method: str, params: ParamsType) \ 

336 -> bytes: # pylint: disable=too-many-return-statements 

337 """ 

338 Translate request parameters to API conform format 

339 

340 .. note:: The build service does not handle URL-encoded Unicode well. 

341 Therefore, parameters are encoded as ``bytes``. 

342 

343 .. warning:: The build service does not declare its parameters properly and developers do 

344 `not intend to fix`_ this server-side. If you want to use _boolean_ parameters, 

345 make sure to use ``True`` and ``False``. If you use ``0`` or ``1`` instead, you 

346 might receive unexpected results. 

347 

348 .. _not intend to fix: https://github.com/openSUSE/open-build-service/issues 

349 /9715 

350 

351 :param params: Request parameter 

352 :param url: URL to which the parameters will be sent 

353 :type url: str 

354 :param method: HTTP method to send request 

355 :type method: str 

356 :return: converted data ready to be used in HTTP request 

357 :rtype: dict or bytes 

358 

359 .. versionchanged:: 0.7.3 

360 

361 Added the ``url`` and ``method`` parameters 

362 

363 .. versionchanged:: 0.9.0 

364 

365 Instances of ``ObjectifiedElement`` are accepted for argument ``params`` 

366 """ 

367 if isinstance(params, bytes): 

368 return params 

369 

370 if isinstance(params, str): 

371 return params.encode('utf-8') 

372 

373 if isinstance(params, StringIO): 

374 params.seek(0) 

375 return params.read().encode('utf-8') 

376 

377 if isinstance(params, (BufferedReader, BytesIO)): 

378 params.seek(0) 

379 return params 

380 

381 if isinstance(params, ObjectifiedElement): 

382 return tostring(params, encoding="utf-8", xml_declaration=True) 

383 

384 if not isinstance(params, dict): 

385 return {} 

386 

387 # The OBS API has a weird expectation regarding boolean parameters and the maintainers have 

388 # made it clear, that they are not going to clean up the API :( 

389 # See: https://github.com/openSUSE/open-build-service/issues/9715 

390 # Also, there are parameters giving the impression that they are boolean, but actually are 

391 # not. 

392 boolean_params = self.get_boolean_params(url=url, method=method) 

393 unexpected_bools = {key for key, value in params.items() 

394 if isinstance(value, bool) and key not in boolean_params} 

395 if unexpected_bools: 

396 for key in unexpected_bools: 

397 params[key] = '1' if params[key] else '0' 

398 

399 return "&".join( 

400 quote(str(key)) 

401 if key in boolean_params 

402 else f"{quote(str(key))}={quote(str(value))}" 

403 for key, value in ( 

404 (key, value) 

405 for key, value in params.items() 

406 if not (key in boolean_params and value in [False, "0", 0, None, ""]) 

407 ) 

408 if value is not None 

409 ).encode()

411 def download(self, url: str, destdir: Path, destfile: typing.Optional[str] = None, 

412 overwrite: bool = False, **params: ParamsType) -> Path: 

413 """ 

414 Shortcut for a streaming GET request 

415 

416 :param str url: Download URL 

417 :param pathlib.Path destdir: Destination directory 

418 :param str destfile: Target file name. If not specified, it will be taken from the URL 

419 :param bool overwrite: switch to overwrite existing downloaded file 

420 :param params: Additional query params 

421 :return: absolute path to file or ``None`` 

422 

423 .. versionadded:: 0.7.0 

424 """ 

425 destdir = destdir if isinstance(destdir, Path) else Path(destdir) 

426 if not destfile: 

427 parsed = urlparse(url) 

428 destfile = Path(parsed.path).name 

429 

430 if destdir.is_file(): 

431 raise OSError(errno.EEXIST, "Target directory is a file", destdir) 

432 

433 target = destdir.joinpath(destfile) 

434 if not overwrite and target.exists(): 

435 raise OSError(errno.EEXIST, "File already exists", target) 

436 if not destdir.exists(): 

437 destdir.mkdir(parents=True, exist_ok=True) 

438 

439 response = self.request(url=url, method="GET", stream=True, params=params) 

440 

441 with target.open("wb") as handle: 

442 for chunk in response.iter_content(1024): 

443 handle.write(chunk) 

444 

445 return target
